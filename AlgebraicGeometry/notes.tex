\section{Affine Algebraic Varieties}
  \begin{definition}
    Let $K$ be a field. Then an Affine Algebraic Variety is a set of the form
    \[\mathcal{V}(I)=\{(a_1,a_2,\dots,a_n)\in K^n\mid f(a_1,a_2,\dots,a_n)=0, \forall f\in I\},\]
    where $I$ is an ideal $I\vartriangleleft K[x_1,x_2,\dots,x_n]$.
  \end{definition}
  \begin{proposition}
    Let $I_1,I_2$ be ideals of $K[x_1,\dots,x_n]$ be such that $I_1\subseteq I_2$, then $\mathcal{V}(I_1)\supseteq\mathcal{V}(I_2)$.
  \end{proposition}
  \begin{example}$ $
    \begin{enumerate}
      \item If $I=\{0\}$, then $\mathcal{V}(I)=K^n$.
      \item If $I=K[x_1,x_2,\dots,x_n]$, then $\mathcal{V}(I)=\{\emptyset\}$.
      \item If $I=\langle x^2+y^2-1\rangle\vartriangleleft\mathbb{R}[x,y]$, then $\mathcal{V}(I)$ is the circle of radius $1$ centred at the origin defined by the equation $x^2+y^2=1$ in $\mathbb{R}^2$.
      \item Let $f_1,f_2,\dots,f_r$ be polynomials of degree $1$ in $x_1,x_2,\dots,x_n$, and let $I=\langle f_1,f_2,\dots,f_r\rangle$ be the ideal generated byu them. Then $\mathcal{V}(I)$ is the set of solutions of the system of linear equations $f_1=f_2=\cdots=f_r=0$. Such a variety is called an affine subspace of $K^n$. This, in particular, includes sets containing a single point. If $P=(a_1,a_2,\dots,a_n)\in K^n$, then the ideal $I=\langle x_1-a_1,x_2-a_2,\dots,x_n-a_n\rangle$ defines the variety $\mathcal{V}(I)=\{P\}$.
      \item Let $I=\langle y^2-x^3\rangle\vartriangleleft\mathbb{R}[x,y]$, then $\mathcal{V}(I)$ is the cuspoidal cubic curve defined by $y^2=x^3$, also know as Neile's semicubal parabol, then $\mathcal{V}(I)$ is the cuspoidal cubic curve defined by $y^2=x^3$, also know as Neile's semicubal parabol, then $\mathcal{V}(I)$ is the cuspoidal cubic curve defined by $y^2=x^3$, also know as Neile's semicubal parabola.
      \item Let $I=\langle y^2-x^3-x^2\rangle\mathbb{R}[x,y]$, then $\mathcal{V}(I)$ is the nodal cubic curve defined by the equation $y^2=x^3+x^2$.
      \item If $I=\langle x,y\rangle\vartriangleleft\mathbb[x,y]$, then $\mathcal{V}(I)=\{(0,0)\}\subset\mathbb{R}^2$.
      \item If $I=\langle x^2,y^2\rangle\vartriangleleft\mathbb[x,y]$, then $\mathcal{V}(I)=\{(0,0)\}\subset\mathbb{R}^2$.
      \item If $I=\idealgen{x^2+y^2+z^2-1,x^2-x+y^2}\idealof\mathbb{R}[x,y,z]$, then $\mathcal{V}(I)\subset\mathbb{R}^3$ is a figure of eight curve from an intersection of the cylinder, $y^2=x^2-x$, and the sphere of radius $1$ at the origin.
      \item The special linear group $SL(n\mathbb{R})$ is also an affine algebraic variety. The set of all $n\times n$ real matrices is $\mathbb{R}^{n^2}$, the determinant is a polynomial in the entries of the matrix, and therefore $SL(n,\mathbb{R})=\mathcal{V}(\idealgen{\det{(A)}-1})$.
    \end{enumerate}
  \end{example}
  \begin{proposition}
    Let $K$ be a field. Let $f_1,f_2,\dots,f_r\in K[x_1,x_2,\dots,x_n]$ and let $I=\idealgen{f_1,f_2,\dots,f_r}$. Then
    \[\mathcal{V}(I)=\{(a_1,a_2,\dots,a_n)\in K^n\mid f_i(a_1,a_2,\dots,a_n)=0, \forall 1\le i\le r\}.\]
  \end{proposition}
  \begin{proof}
    We know $f_i\in I$ for each $i$, so if $(a_1,\dots,a_n)\in\mathcal{V}(I)$, then $f_i(a_1,\dots,a_n)=0$ for every $i$.
    Conversely, assume that $f_i(a_1,\dots,a_n)=0$ for every $i$. Any $f\in I$ can be written as a $K[x_1,\dots,x_n]$-linear combination of the $f_i$, so
    \[f(a_1,\dots,a_n)=\sum_{i=1}^rf_i(a_1,\dots,a_n)g_i(a_1,\dots,a_n)=0,\]
    so $(\seq{a}{n})$.
  \end{proof}
  \begin{definition}
    A ring $R$ is called Noetherian if all its ideals can be generated by finitely many elements.
  \end{definition}
  \begin{theorem}[Hilbert Basis Theorem]
    If $K$ is a field, then the polynomial ring $K[\seq{x}{n}]$ is Noetherian for any $n\ge0$.
  \end{theorem}
  \begin{example}$ $
    \begin{enumerate}
      \item If $R$ is Noetherian and $I\vartriangleleft R$, then $R/I$ is also Noetherian.
      \item $\mathbb{Z}$, the ring of integers, is Noetherian because it is a PID, so every ideal is generated by one element.
      \item $K[x_1,x_2,\dots]$, the polynomial rinf over the field $K$ in finitely many variables is an example of a non-Noetherian ring, because the ideal $\idealgen{x_1,x_2,\dots}$ is not finitely generated.
    \end{enumerate}
  \end{example}
  \begin{proposition}\label{1.8}$ $
    \begin{enumerate}
      \item Let $V_1=\mathcal{V}(I_1),\dots,V_k=\mathcal{V}(I_k)$ be affine algebraic varieties in $K^n$. Then
        \[V_1\cup\cdots\cup V_k=\mathcal{V}(I_1\cap\cdots\cap I_k)=\mathcal{V}(I_1I_2\cdots I_k)\]
        is also an affine algebraic variety.
      \item Let $V_\alpha=\mathcal{V}(I_\alpha),\alpha\in A$ be affine algebraic varieties in $K^n$. Then
        \[\bigcap_{\alpha\in A}V_\alpha=\mathcal{V}\left(\sum_{\alpha\in A}I_\alpha\right)\]
        is also an affine algebraic variety.
      \item Let $V_1=\mathcal{V}(I_1)\subseteq K^m, V_2=\mathcal{V}(I_2)\subseteq K^n$ be affine algebraic varieties. Then $V_1\times V_2\subseteq K^m\times K^n\cong K^{m+n}$ is also an affine algebraic variety.
    \end{enumerate}
  \end{proposition}
  \begin{proof}$ $
    \begin{enumerate}
      \item 
        We first assume that $k=2$. Let $P\in V_1\cup V_2$ and let $f\in I_1\cap I_2$. If $P\in V_1$, then $f(P)=0$ because $f\in I_1$, and similarly if $P\in V_2$ then $f(P)=0$ because $f\in I_2$. Therefore in either case we have $f(P)=0$, so $P\in\mathcal{V}(1_1\cap I_2)$, so $V_1\cup V_2\subseteq\mathcal{V}(I_1\cap I_2)$.\\
        We also know $I_1I_2\subseteq I_1\cap I_2$, so we also have $\mathcal{V}(I_1\cap I_2)\subseteq\mathcal{V}(I_1I_2)$.\\
        Let now $P\in K^n\setminus(V_1\cup V_2)$. 
        Then $P\notin V_1$, so there exists $f_1\in I_1$ such that $f_1(P)\ne0$. Similarly, $P\ne V_2$ so there exists $f_2\in I_2$ such that $f_2(P)\ne0$. 
        Then $f_1f_2\in I_1I_2$ and $(f_1 f_2)(P)=f_1(P)f_2(P)\ne0$, therefore $P\notin\mathcal{V}(I_1I_2)$. 
        This implies $\mathcal{V}(I_1I_2)\subseteq V_1\cup V_2$. Hence we obtain
          \[V_1\cup V_2=\mathcal{V}(I_1\cap I_2)=\mathcal{V}(I_1I_2).\]
        For $k>2$ we use induction on $k$.
      \item
        Let $P\in\bigcap\limits_{\alpha\in A}V_{\alpha}$ and let $f=\sum\limits_{\alpha\in A}I_{\alpha}$. Then $f$ can be written as 
        \[f=\sum_{\alpha\in A}f_{\alpha}\]
        with $f_{\alpha}\in I_{\alpha}$ for every $\alpha\in A$ and $f_\alpha=0$ for all but finitely many $\alpha$. We know $P\in V_\alpha$ for every $\alpha\in A$, so $f_{\alpha}(P)=0$ for every $\alpha\in A$, therefore
        \[f(P)=\sum_{\alpha\in A}f_{\alpha}(P)=\sum_{\alpha\in A}0=0.\]
        Hence $P\in\mathcal{V}\left(\sum\limits_{\alpha\in A}I_\alpha\right)$. This implies
        \[\bigcap_{\alpha\in A}V_\alpha\subseteq\mathcal{V}\left(\sum_{\alpha\in A}I_\alpha\right).\]
        Assume now that $P\in K^n\setminus\bigcap\limits_{\alpha\in A}V_\alpha$. Then there exists $\alpha_0\in A$ such that $P\notin V_{\alpha_0}$, therefore there exists $f\in I_{\alpha_0}$ such that $f(P)\ne 0$. Now $f\in\sum\limits_{\alpha\in A}I_\alpha$, therefore $P\notin\mathcal{V}\left(\sum\limits_{\alpha\in A}I_\alpha\right)$. This implies that
        \[\mathcal{V}\left(\sum_{\alpha\in A}I_\alpha\right)\subseteq\bigcap_{\alpha\in A}V_\alpha.\]
      \item 
        Let $\seq{x}{m}$ be coordinates on $K^m$, and $\seq{y}{n}$ be coordinates on $K^n$. Let $J_1=\idealgen{I_1}\idealof K[\seq{x}{m},\seq{y}{n}]$, where $I_1$ is an ideal in $K[\seq{x}{m}]$. We claim that $\mathcal{V}(J_1)=V_1\times K^n$.\\
        Let $P=(\seq{a}{m},\seq{b}{n})\in\mathcal{V}(J_1)$. Then $f(P)=0$ for every $f\in I_1$, since $I_1\subset J_1$. As $f$ only involves the variables $\seq{x}{m}$,
        \[0=f(P)=f(\seq{a}{m},\seq{b}{n})=f(\seq{a}{m}).\]
        Hence $(\seq{a}{m})\in\mathcal{V}(I_1)=V_1$, so $P\in V_1\times K^n$, therefore $\mathcal{V}(J_1)\subseteq V_1\times K^n$.\\
        Now let $P=(\seq{a}{m},\seq{b}{n})\in V_1\times K^n$. Let $f\in J_1$. Then $f$ can be written as $f=\sum\limits_{i=1}^r{f_ig_i}$, where $f_i\in I$ and $g_i\in K[\seq{x}{m},\seq{y}{n}]$. Now
        \[f_i(P)=f_i(a_1,\dots,a_m,\seq{b}{n})=f_i(\seq{a}{m})=0.\]
        Therefore $f(P)=0$, so $P\in\mathcal{V}(J_1)$. This means $V_1\times K^n\subseteq\var{J_1}$, and so we have shown that $\var{J_1}=V_1\times K^n$.
        Similarly, if we define $J_2=\idealgen{I_2}\idealof K[\seq{x}{m},\seq{y}{n}]$, then $\var{J_2}=K^m\times V_2$. Hence
        \[V_1\times V_2=(V_1\times K^n)\cap(K^m\times V_2)=\var{J_1}\cap\var{J_2}=\var{J_1+J_2}.\qedhere\]
    \end{enumerate}
  \end{proof}
  \begin{example}$ $
    \begin{enumerate}
      \item 
        We wish to find an ideal $I\idealof\mathbb{R}[x,y]$ such that $\var{I}={(1,1),(2,-3)}$. Let $I_1=\idealgen{x-1,y-1}$ and $I_2=\idealgen{x-2,y+3}$. Then $\var(I_1)={(1,1)}$ and $\var{I_2}={(2,-3)}$. Hence, by Proposition \ref{1.8}(i),
        \[I=\var{(x-1)(x-2),(x-1)(y+3),(y-1)(x-2),(y-1)(y+3)}\]
        is a suitable ideal.
      \item 
        Let $V_1=\var{\idealgen{x_1^2+x_2^2-1}}\subset\mathbb{R}^2$ and let $V_2=\var{\idealgen{y_1-y_2}}\subset\mathbb{R}^2$. Then $V_1\times V_2=\var{\idealgen{x_1^2+x_2^2-1,y_1-y_2}}\subset\mathbb{R}^4$ is a cylinder.
    \end{enumerate}
  \end{example}

\subsection{Affine Spaces}
  \begin{definition}
    Let $K$ be a field, and let $V$ be a finite dimensional vector space over $K$. Then $U\subseteq V$ is called an affine subspace of $V$ if $U=\emptyset$ or or $U=u_0+W=\{u_0+w\mid w\in W\}$, where $u_0\in U$ and $W$ is a linear subspace of $V$.
  \end{definition}
  \begin{example}$ $
    \begin{enumerate}
      \item $K^n,\emptyset$, and any set containing a single point are affine subspaces of $K^n$.
      \item The line $y=\frac{1}{2}x+1$ is an affine subspace of $\mathbb{R}^2$, in this case we can take $W=span{(2,1)}$, which is the line $y=\frac{1}{2}x$, and $u_0=(0,1)$.
    \end{enumerate}
  \end{example}
  \begin{proposition}
    Let $U=u_0+W$ be a non-empty affine subspace in a vector space $V$. Then the linear subspace $W$ is uniquely deteremined by $U$, and $U$ can be written as $U=u_1+W$ for any $u_1\in U$.
  \end{proposition}
  \begin{proof}
    By definition, $U=u_0+W$, and therefore $W=U-u_0=\{u-u_0\mid u\in U\}$. Let $u_1\in U$ be arbitrary, then $u=u_0+w_1\in W$. We consider the set 
    \[
      U-u_1=\{u-u_1\mid u\in U\}=\{u-u_0-w_1\mid u\in U\}.
    \]
    We have $u-u_0\in U-u_0=W$, therefore $u-u_0-w_1\in W$, as $W$ is a linear subspace. This means that $U-u_1\subseteq W$. Let $w\in W$, then 
    \[
      w+u_1=w+w_1+u_0=(w+w_1)+u_0\in U,
    \]
    so $w\in U-u_1$, therefore $U-u_1=W$. Hence $U=u_1+W$ for any $u_0\in U$.
  \end{proof}
  \begin{proposition}
    Let $U_1,U_2$ be affine subspaces of $V$. Then $U_1\cap U_2$ and $U_1+U_2=\{u_1+u_2\mid u_1\in U_1, u_2\in U_2\}$ are also affine subspaces of $V$.
  \end{proposition}
  \begin{proof}
    If $U_1\cap U_2=\emptyset$, then $U_1\cap U_2$ is an affine subspace, so assume that $U_1\cap U_2\ne\emptyset$ and let $u_0\in U_1\cap U_2$. 
    So $U_1=u_0+W_1, U_2=u_0+W_2$, where $W_1,W_2$ are vector subspaces of $V$ and $U_1\cap U_2=u_0+(W_1\cap W_2)$. THe intersection of linear subspaces is also a linear subspace, therefore $U_1\cap U_2$ is an affine subspace of $V$.\\
    If either $U_1=\emptyset$ or $U_2=\emptyset$, then $U_1+U_2=\emptyset$, so it too is an affine subspace of $V$.\\
    If neither $U_1$ nor $U_2$ is the empty set, then we can write them as $U_1=v_1+W_1,U_2=v_2+W_2$, where $v_1\in U_1$ and $v_2\in U_2$ and $W_1,W_2$ are linear subspaces of $V$. We have
    \begin{align*}
      U_1+U_2=&\{u_1+u_2\mid u_1\in U_1, u_2\in U_2\}\\
              &\{(v_1+w_1)+(v_2+w_2)\mid w_1\in W_1, w_2\in W_2\}\\
              &\{(v_1+v_2)+w\mid w\in W_1+W_2\}.
    \end{align*}
    $W_1+W_2$ is also a linear subspace of $V$, therefore $U_1+U_2$ is an affine subspace.
  \end{proof}
  \begin{definition}
    The dimension of an affine subspace $U=u_0+W$ is defined to be $\dim{W}$. For convention we write $\dim{\emptyset}=-1$.
  \end{definition}
  \begin{definition}
    Let $K$ be a field. $K^n$, considered as an affine subspace of itself, is called an $n$-dimensionsal affine space over $K$, and is denoted by $\mathbb{A}^n(K)$ or $\mathbb{A}^n$ if the field is understood.
  \end{definition}
  \begin{definition}
    Let $V_1,V_2$ be vector spaces. An affine map from $V_1$ to $V_2$ is defined to be a function $\varphi:V_1\to V_2$ which can ve written in the form $\varphi(x):V_1\to V_2$ is a linear transformation and $b\in V_2$.
  \end{definition}
  \begin{definition}
    Let $X,Y$ be subsets of a vector space $V$. $X$ and $Y$ are called affine equivalent if there exists mutually inverse affine maps $\varphi,\psi:V\to V$ such that $\varphi(X)=Y$ and $\psi(Y)=X$.
  \end{definition}
\subsection{The Correspondence Between Varieties and Ideals}
  \begin{definition}
    Let $K$ be a field. Let $X\subseteq \mathbb{A}^n(K)$. The ideal of $X$ is defined as 
    \[
      \mathcal{I}(X)=\{f\in K[x_1,\dots,x_n]\mid f(a_1,\dots,a_n)=0,\forall (a_1,\dots,a_n)\in X\}.
    \]
  \end{definition}
  $\mathcal{I}(X)$ is indeed an ideal of $K[x_1,\dots,x_n]$.
  \begin{proposition}
    Let $K$ be a field.
    \begin{enumerate}
      \item If $J\idealof K[x_1,\dots,x_n]$, then $J\subset \mathcal{I}(\mathcal{V}(J))$.
      \item $X\subseteq \mathcal{V}(\mathcal{I}(X))$ for any $X\in \mathbb{A}^n(K)$. Equality holds if and only if $X$ is an affine algebraic variety.
    \end{enumerate}
  \end{proposition}
  \begin{proof}$ $
    \begin{enumerate}
      \item Let $f\in J$. Then $f(P)=0$ for every $P\in \mathcal{V}(J)$ by the definition of $\mathcal{V}$, so $f\in \mathcal{I}(\mathcal{V}(J))$ by the definition of $\mathcal{I}$, therefore $J\subseteq \mathcal{I}(\mathcal{V}(J))$ as claimed. \qedhere
    \end{enumerate}
  \end{proof}
  \begin{example}
    Let $J=\left\langle x^2,y^2\right\rangle\idealof \mathbb{R}[x,y]$, then $\mathcal{V}(J)=\{(0,0)\}$, so $\mathcal{I}(\mathcal{V}(J))=\left\langle x,y\right\rangle\ne J$.
  \end{example}
  \begin{definition}
    Let $R$ be a commutative ring. Let $I\idealof R$ be an ideal. The radical of $I$, denoted by $\sqrt{I}$, is 
    \[
      \sqrt{I}=\{a\in R\mid\exists n\in \mathbb{N} \text{ such that }a^n\in I\}.
    \]
    $I$ is called a radical ideal if $\sqrt{I}=I$.
  \end{definition}
  \begin{example}
    $\sqrt{\left\langle x^2,y^2\right\rangle}=\left\langle x,y\right\rangle$, but in general, one has to consider the roots of the ideal, not just the generators.
  \end{example}
  \begin{definition}
    A field $K$ is called algebraically closed if every polynomial of positive degree in $K[x]$ has a root in $K$.
  \end{definition}
  \begin{theorem}[Hilbert's Nullstellensatz]
    Let $K$ be an algebraically closed field.
    \begin{enumerate}
      \item Any maximal ideal of $K[x_1,\dots,x_n]$ is of the form \[\left\langle x_1-a_1,\dots,x_n-a_n\right\rangle,\] where $a_i\in K, 1\le i\le n$.
      \item If $J$ is a proper ideal of $K[x_1,\dots,x_n]$, then $\mathcal{V}(J)\ne\emptyset.$
      \item $\mathcal{I}(\mathcal{V}(J))=\sqrt{J}$ for any $J\idealof K[x_1,\dots,x_n].$
    \end{enumerate}
  \end{theorem}
  From the Nullstellensatz, we have that $\mathcal{V}$ and $\mathcal{I}$ are mutual inverse bijections of eachother between radical ideals and algebraic varieties:

\subsection{Irreducibility}
  \begin{definition}
    An affine algebraic variety is said to be reducible if there exist affine algebraic varieties $V_1,V_2$ such that $V_1\ne V\ne V_2$ and $V=V_1\cup V_2$. If $V$ is not reducible, then it is called irreducible.
  \end{definition}
  \begin{example}$ $
    \begin{enumerate}
      \item The variety $\mathcal{V}\left(\left\langle x^2-y^2\right\rangle\right)\subset \mathbb{R}^2$ is reducable, since it is the union of $\mathcal{V}\left(\left\langle x-y\right\rangle\right)$ and $\mathcal{V}\left(\left\langle x+y\right\rangle\right)$, which are the lines $y=x$ and $y=-x$.
      \item Any variety consisting of a single point is irreducible.
      \item If $K$ is infinite, the affine line $\mathbb{A}^1$ is irreducible, since any subvariety is either a finite set or the whole of $\mathbb{A}^1$.
    \end{enumerate}
  \end{example}
  \begin{theorem}
    Every affine algebraic variety can be decomposed into a union $V=V_1\cup\cdots\cup V_k$ such that every $V_i,1\le i\le k$, is an irreducible affine algebraic variety and $V_i\not\subseteq V_j$ for $i\ne j$. the decomposition is unique up to the order of terms. We call the $V_i$ the irreducible components of $V$. 
  \end{theorem}
  \begin{proof}
    We call an affine algebraic variety good if it can be written as the union of finitely many irreducible affine algebraic varieties and bad otherwise.\\
    Assume that $V$ is a bad affine algebraic variety. $V$ cannot be irreducible, so $V=U_1\cup W_1$ for some proper subvarieties. Furthermore, at least one of $U_1$ and $W_1$ is bad, since if both of them were good, their union would also be good. Without loss of generality, assume $U_1$ is bad. Then, by a similar arguement, $U_1=U_2\cup W_2$, where $U_2$ and $W_2$ are both proper subvarieties of $U_1$, and at least one of them is bad, which we may assume to be $U_2$. Continuing this, we obtain an infinite strictly decreasing sequence $U_1\supsetneq U_2\supsetneq U_3\supsetneq\cdots$ of bad varieties. We know that $U_k\supsetneq U_{k+1}$ implies $\mathcal{I}(U_k)\subseteq\mathcal{I}(U_{k+1})$. As $U_k$ and $U_{k+1}$ are both affine algebraic varieties, $U_k=\mathcal{V}(\mathcal{I}(U_k))$ and $U_{k+1}=\mathcal{V}(\mathcal{I}(U_{k+1}))$. $\mathcal{V}(\mathcal{I}(U_{k}))\ne\mathcal{V}(\mathcal{I}(U_{k+1}))$, so $\mathcal{I}(U_k)\ne\mathcal{I}(U_{k+1})$. Hence, $\mathcal{I}(U_1)\subsetneq\mathcal{I}(U_2)\subsetneq\mathcal{I}(U_3)\subsetneq\cdots$ is a strictly increasing sequence of ideals, contradicting the Noetherian property of the polynomial ring, hence $V$ is not bad.\\
    Therefore any affine algebraic variety is goos, and can be written as the union of finitely many irreducible affine algebraic varieties. By discarding the varieties which are contained in others, we can write $V=V_1\cup\cdots\cup V_k$ such that for every $V_i,1\le i\le k$, is an irreducible affine algebraic variety and $V_i\not\subseteq V_j$ for $i\ne j$. This proves the existence of the decomposition.\\
    Let $V=V_1^\prime\cup\dots\cup V_\ell^\prime$ be anpther decomposition of $V$ into irreducible varieties with $V_i^\prime\not\subseteq V_j^\prime$ for $i\ne j$. Then
    \[V_1=V_1\cap V=(V_1\cap V_1^\prime)\cup\cdots\cup(V_1\cap V_\ell^\prime).\]
    As $V_1$ is irreducible, there exists $i$ such that $V_1=V_1\cap V_i^\prime$, that is $V_1\subseteq V_i^\prime$. By applying this argiment to $V_i^\prime$, we obtain that there exists $j$ such that $V_i^\prime\subseteq V_j$, therefore $V_1\subseteq V_j$. This implies $j=1$, and therefore $V_1=V_i^\prime$.
    We apply this arguement to each $V_r,1\le r\le k$, and $V_s^\prime,1\le s\le\ell$, we obtain a bijection between $\{V_1,\dots,V_k\}$ and $\{V_1^\prime,\dots,V_\ell^\prime\}$, so the decomposition is unique up to ordering.
  \end{proof}
  \begin{definition}
    Let $R$ be a commutative ring. An ideal $I\idealof R$ is called a prime ideal if $I\ne R$ and for every $a,b\in R,ab\in I$ implies $a\in I$ or $b\in I$.
  \end{definition}
  \begin{example}$ $
    \begin{enumerate}
      \item $\left\langle n\right\rangle=n \mathbb{Z}\idealof \mathbb{Z}$ is a prime ideal if and only if $n$ is a prime number or $n=0$.
      \item Let $K$ be a field and let $f\in K[x_1,\dots x_n]$. The ideal $\left\langle f\right\rangle$ is a prime ideal of $K[x_1,\dots,x_n]$ if and only if $f$ is an irreducible polynomial or $f=0$.
      \item Let $I=\left\langle x^2-1-y,x^2-1+y\right\rangle\idealof \mathbb{R}[x,y]$. We know $x^2-1-y$ and $x^2-1+y$ are irreducible polynomials. We have
        \[\frac{x^2-1-y}{2}+\frac{x^2-1+y}{2}=x^2-1=(x-1)(x+1)\in I,\]
        and $x-1,x+1\notin I$, so $I$ is not a prime ideal of $\mathbb{R}[x,y]$.
    \end{enumerate}
  \end{example}
  \begin{proposition}
    An affine algebraic variety $W\subseteq\mathbb{A}^n(K),W\ne\emptyset$, is irreducible if and only if $\mathcal{I}(W)$ is not prime.
  \end{proposition}
  From this proposition, and from the Nullstellensatz be have that $\mathcal{V}$ and $\mathcal{I}$ are mutually inverse bijections between prime ideals and non-empty irreducible affine algebraic varieties:

      \arrow[r, "\mathcal{V}", shift left=1.5ex]
      \arrow[r, "\mathcal{I}", leftarrow, shift right=1.5ex]
      &
      \left\{\parbox{5cm}{\centering Non-empty irreducible affine\\ algebraic varieties in $\mathbb{A}^n$}\right\}.
    \end{tikzcd}
  \]

    Let $n\ge2$ be an integer, let $K$ be a field and let $p\in K[x_1,\dots,x_n]$ be a polynomial which does not involve the variable $x_i$ for some $i, 1\le i\le n$, and which is not the square of another polynomial in $K[x_1,\dots,x_n]$. Then $x_i^2-p$ is irreducible in $K[x_1,\dots,x_n]$.

\section{Morphisms and Co-ordinate Rings}
  \begin{definition}
    Let $V\subseteq \mathbb{A}^n(K)$ be an affine algebraic variety. Its co-ordinate ring, denoted by $K[V]$, is defined to be 
    \[K[V]=K[x_1,\dots,x_n]/\mathcal{I}(V).\]
  \end{definition}
  \begin{example}$ $
    \begin{enumerate}
      \item Let $K$ be an infinite field, then $\mathcal{I}(\mathbb{A}^n)=\{0\}$, so $K[\mathbb{A}^n]=K[x_1,\dots,x_n]$.
      \item Let $V=\mathcal{V}\left(\left\langle x^2+y^2-1\right\rangle\right)\subseteq \mathbb{C}^2$. $x^2+y^2-1$ is irreducible, so $\mathcal{I}(V)=\left\langle x^2+y^2-1\right\rangle$ and $\mathbb{C}[V]=\mathbb{C}[x,y]/\left\langle x^2+y^2-1\right\rangle$
    \end{enumerate}
  \end{example}
  \begin{definition}
    Let $V\subseteq \mathbb{A}^m(K)$ and $W\subseteq\mathbb{A}$ be affine algebraic varieties. A morphism $\phi:V\to W$ is a function of the form $\phi(P)=(\phi_1(P),\dots,\phi_n(P))$ with each $\phi_i\in K[V]$ for each $i,1\le i\le n$.
  \end{definition}
  Elements of $K[V]$ can be considered as morphisms $V\to \mathbb{A}^1$.
  The composition of two morphisms $\phi:V\to W$ and $\psi:W\to X$ is a morphism $\psi\circ\phi:V\to X$.
  \begin{definition}
    The morphism $\phi:V\to W$ is called an isomorphism if it has an inverse morphism, that is, if there  exists a morphism $\psi:W\to V$ such that $\psi\circ\phi=\id[V]$ and $\phi\circ\psi=\id[W]$, where $\id[X]:X\to X$ is the identity finction for a set $X$. $V$ and $W$ are called isomorphic if there exists an isomorphism between them.
  \end{definition}
  \newpage
  \begin{example}$ $
    \begin{enumerate}
      \item 
        Any affine map is a morphism. Invertible affine maps are isomorphisms. Affine equivalent affine algebraic varieties are isomorphic.
      \item 
        Let $K$ be an arbitrary infinite field. Let $t$ be the co-ordinate on $\mathbb{A}^1$ and $x,y$ the co-ordinates on $\mathbb{A}^2$. Let $V=\mathbb{A}^1$ and $W\subseteq \mathbb{A}^2$ be the parabola defined by the equation $y-x^2=0$. Then $\mathcal{I}(V)=\{0\}, \mathcal{I}(W)=\left\langle y-x^2\right\rangle$, so $K[V]=K[t]$ and $K[W]=K[x,y]/\left\langle y-x^2\right\rangle$.\\
        $\phi:V\to W, t\mapsto(t,t^2)$ is a morphism. $t$ and $t^2$ are elements of $K[V]$ and $x=t,y=t^2$ satisfy $y-x^2=0$ for any $t$.
        $\psi:W\to V, (x,y)\to x$ is also a morphism. We know $x\in K[W]$ and have no equations to check. Then
        \[(\psi\circ\phi)(t)=\psi(t,t^2)=t,\quad(\phi\circ\psi)(x,y)=\phi(x)=(x,x^2)=(x,y).\]
        Hence $\phi$ and $\psi$ are inverses of eachother, so $V$ and $W$ are isomorphic.
      \item 
        Let $K$ be an infinite field of characteristic other than 2, for example $K=\mathbb{R}$ or $\mathbb{C}$. Let $V=\mathbb{A}^1$ and let $W\subseteq\mathbb{A}^2$ be the nodal curve defined by the equation $x^3+x^2-y^2=0$. $\phi:V\to W,t\mapsto(t^2-1,t(t^2-1))$ is a morphism. $t^2-1$ and $t(t^2-1)$ are element of $K[V]$ and $x=t^2-1,y=t(t^2-1)$ satisfy $x^3+x^2-y^2=0$ for any $t$, so $\phi$ is function $V\to W$. $\phi$ is not an isomorphism, since it is not injective, $\phi(1)=\phi(-1)=(0,0)$.
      \item 
        Let $H\subseteq \mathbb{A}^2$ be the hyperbola defined by $xy-1=0$, and let $\phi:H\to \mathbb{A}^1$ be the morphism $(x,y)\mapsto x$. $\phi$ is not an isomorphism, since it is not surjective because $0\notin\phi(H)$.
      \item 
        Let $V=\mathbb{A}^1$ and let $W=\mathcal{V}(\left\langle y^2-x^3\right\rangle)\subseteq \mathbb{A}^2$ be the cuspidal cubic curve. $\phi:V\to W, t\mapsto(t^2,t^3)$ is a morphism. $t^2$ and $t^3$ are elements of $K[V]$ and $x=t^2,y=t^3$ satisfy $y^2-x^3=0$ for any $t$, so $\phi$ is a morphism $V\to W$. $\phi$ is bijective as a fuction, but it is not an isomorphism. It has an inverse as a function
        \[\psi:W\to V,\quad\psi(x,y)=\begin{cases}\frac{y}{x},&\text{if }x\ne 0,\\ 0,&\text{if }x=0,\end{cases}\]
        but this is not a morphism.
      \item
        Let $x,y$ be the co-ordinates on $\mathbb{A}^2(\mathbb{C})$ and $u,v,w$ the co-ordinates on $\mathbb{A}^3(\mathbb{C})$. Let $V=\mathbb{A}^2$ and $W\subseteq \mathbb{A}^3$ be the variety given by the equation $uw-v^2=0$, a double cone. $\phi:V\to W,(x,y)\mapsto(x^2,xy,y^2)$ is a morphism. $x^2,xy,y^2$ are elements of $K[V]$ and $u=x^2,v=xy$ and $w=y^2$ satisfy
        \[uw-v^2=x^2y^2-(xy)^2=0\]
        for all $x,y$, therefore $\phi(V)\subseteq W$, so $\phi$ is indeed a morphism $V\to W$.
      \item 
        Let $V=\mathbb{A}^1(\mathbb{R})$ and let $W=\mathcal{V}\left(\left\langle x^2+y^2-1\right\rangle\right)\subseteq \mathbb{A}^2(\mathbb{R})$ be the unit circle. $\phi:V\to W,\phi(t)=(\cos{t},\sin{t})$ is a function from $V$ to $W$, but it is not a morphism, as $\cos{t}$ and $\sin{t}$ are not elements of $\mathbb{R}[\mathbb{A}^1]\cong \mathbb{R}[t]$.
    \end{enumerate}
  \end{example}
  \begin{theorem}
    Let $V\subseteq{A}^m(K),V\ne\emptyset$, and $W\subseteq \mathbb{A}^n(K)$ be affine algebraic varieties. For a morphism $\phi:V\to W$, define $\phi^*:K[W]\to K[V], \phi^*(f)=f\circ\phi$. Here, $^*:\phi\mapsto\phi^*$ is a bijection between morphisms of affine algebraic varieties $\phi:V\to W$ and ring homomorphisms $K[W]\to K[V]$ preserving $K$.
  \end{theorem}
  \begin{proof}
    Let $f\in K[W]$. Let $y_1,\dots,y_n$ be the co-ordintates on $\mathbb{A}^n$. We know $f=F+\mathcal{I}(W)$ for some $F\in K[y_1,\dots,y_n]$. Now,
    \[\phi^*(f)=f\circ\phi=F\circ\phi=F(\phi_1,\dots,\phi_n)\]
    where $\phi_1,\dots,\phi_n\in K[V]$ are the components of $\phi$. As $F$ is a polynomial, $\phi^*(f)=F(\phi_1,\dots,\phi^n)\in K[V]$. An element of $K[V]$ is determined by its values as a finction $V\to K$, so $\phi^*(f)$ is a well-defined element of $K[V]$. This shows that $\phi^*$ is indeed a function $K[W]\to K[V]$.\\
    We know $\phi^*(\lambda)=\lambda$ for any $\lambda\in K$, and that $\phi^*(f+g)=\phi^*(f)+\phi^*(g)$ and $\phi^*(fg)=\phi^*(f)\phi^*(g)$ for all $f,g\in K[W]$ by the definition of $\phi^*$, and so $\phi^*$ is a homomorphism of rings which preserve $K$.\\
    Given a homomorphism $\alpha:K[W]\to K[V]$ with $\alpha(\lambda)=\lambda$ for every $\lambda\in K$, let $\phi_i=\alpha(y_i+\mathcal{I}(W))\in K[V]$ for $i=1,\dots,n$. Then $\phi=(\phi_1,\dots,\phi_n)$ is clearly a morphism $V\to \mathbb{A}^n$. We need to prove that $\phi(V)\subseteq W$, so that $\phi$ is a mophism $V\to W$.\\
    Let $P\in V$ an aribitrary point. In order to prove $\phi(P)\in W$, we need to show $g(\phi(P))=0$ for any $g\in\mathcal{I}(W)$.\\
    Let $q:K[y_1,\dots,y_n]\to K[W]=K[y_1,\dots,y_n]/\mathcal{I}(W)$ be the quotient map $q(f)=f+\mathcal{I}(W)$. Then $\phi_i=\alpha(q(y_i))$, so 
    \[g(\phi(P))=g(\alpha(q(y_1))(P),\dots,\alpha(q(y_n))(P)).\]
    $g$ is a polynomial, composed of its variables, scalars in $K$, addition and multiplication, while $\alpha$ and $q$ are ring homomorphisms preserving $K$, therefore 
    \[g(\alpha(q(y_1)),\dots,\alpha(q(y_n)))=\alpha(q(g(y_1,\dots,y_n)))=\alpha(0)=0\]
    since $g\in \mathcal{I}(W)=\ker{q}$. If we substitute the co-ordinates of $P$, we obtain $g(\phi(P))=0$. As this holds for every $g\in \mathcal{I}(W)$, we have $\phi(P)\in W$, so $\phi$ is indeed a morphism $V\to W$.\\
    If $\phi=(\phi_1,\dots,\phi_n)$ is a morphism of algebraic varieties $V\to W$, then $\phi_i=\phi^*(y_i+\mathcal{I}(W))$, so the ring homomorphism constructed from $\phi^*$ is exactly $\phi$. Conversely, if $\alpha$ is a ring homomorphism $\alpha:K[W]\to K[V]$ and $\phi:V\to W$ is the morphism of algebraic varieties constructed from $\alpha$, then $\alpha=\phi^*$.\\
    This shows that the constructions associating a ring homomorphism to a homomorphism of algebraic varieties and a homomorphism of algebraic varieties to a ring homomorphism are inverses of each other, therefore $^*:\phi\mapsto\phi^*$ is a bijection.
  \end{proof}
  \begin{corollary}
    $\phi:V\to W$ is an isomorphism of affine algebraic varieties if and only if $\phi^*:K[W]\to K[V]$ is an isomorphism of rings. $V$ and $W$ are isomorphic if and only if there exists an isomorphism between $K[V]$ and $K[W]$ which preverses $K$.
  \end{corollary}
  \begin{definition}
    An element $r$ in a ring is called nilpotent if there exists a positive integer $n$ such that $r^n=0$.
  \end{definition}
  \begin{example}
    $\left\langle 8\right\rangle\idealof \mathbb{Z}$ is not a radical ideal, as $r\notin \left\langle 8\right\rangle$, but $2^3\in \left\langle 8\right\rangle$. The quotient ring $\mathbb{Z}_8=\mathbb{Z}/\left\langle 8\right\rangle$ has nilpotent elements other than $0$, for example, $(\bar{2})^3=\bar{0}$.
  \end{example}
  \begin{definition}
    Let $K$ be a field. A ring $R$ is called $K$-algebra if it contains $K$ as a subring. A ring homomorphism $R\to S$ between $K$-algebra homomorphism if it preserves $K$. A $K$-algebra $R$ is called finitely generated if it can be generated as a ring by $K$ and finitely many other elements.
  \end{definition}
  \begin{example}$ $
    \begin{enumerate}
      \item A polynomial ring over $K$ is a $K$-algebra.
      \item Co-ordinate rings of non-empty affine algebraic varieties over a field $K$ are finitely generated $K$-algebras.
    \end{enumerate}
  \end{example}
  If $K$ is algebraically closed, co-ordinate rings of non-empty affine algebraic varieties over $K$ are exactly the finitely generated $K$-algebras with no nilpotent elements other than $0$. Hence, we have a bijection
    \[\begin{tikzcd}[column sep=huge]
      \left\{\parbox{3.5cm}{\centering Isomorphism classes\\ of non-empty\\affine algebraic\\varieties over $K$}\right\}
      \arrow[r, "{V\mapsto K[V]}"]&
      \left\{\parbox{4cm}{\centering Isomorphism classes\\ of finitely generated\\ $K$-algebras with no\\ nilpotents other than $0$}\right\}.
    \end{tikzcd}\]
